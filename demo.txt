git checkout -- <file> : dischange file which is not on stagging area (before add)
git diff : see the different 
git reset <file> : unstage
git checkout -b <branch_name> : mk new branch and checkout to that branch
git reset --soft <commit_ip> : delete any commit after <file>
git reset --mixed <commit_ip> : delete any commit after <file> & unstage that file
git reset --hard <commit_ip> : delete any commit & changes after <file> & unstage that file
git revert <commit_ip> : unchange that file
git push : push new commit to remote repo
git clone <linkgithub> : use to pull a repo to new user
git pull : pull new commit to your pc
